 81:     /**

 88:      * @return array
 89:      */
 90:     public function getHeaders()
 91:     {

 93:         $headers = array();
 94: 
 95:         foreach ($this->attributes as $key => $value) {
 96:             // Does our server attribute have our header prefix?
 97:             if (self::hasPrefix($key, self::$http_header_prefix)) {
 98:                 // Add our server attribute to our header array
 99:                 $headers[
100:                     substr($key, strlen(self::$http_header_prefix))
101:                 ] = $value;
102: 
103:             } elseif (in_array($key, self::$http_nonprefixed_headers)) {
104:                 // Add our server attribute to our header array

106:             }
107:         }
