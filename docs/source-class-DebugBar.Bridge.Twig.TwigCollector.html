23:  *
24:  * <code>
25:  * $env = new TraceableTwigEnvironment(new Twig_Environment($loader));
26:  * $debugbar->addCollector(new TwigCollector($env));
27:  * </code>
28:  */
29: class TwigCollector extends DataCollector implements Renderable
30: {
31:     public function __construct(TraceableTwigEnvironment $twig)
32:     {
33:         $this->twig = $twig;
34:     }
35: 
36:     /**
37:      * {@inheritDoc}
38:      */
39:     public function collect()
40:     {
41:         $templates = array();
42:         $accuRenderTime = 0;
43: 
44:         foreach ($this->twig->getRenderedTemplates() as $tpl) {
45:             $accuRenderTime += $tpl['render_time'];
46:             $templates[] = array(
47:                 'name' => $tpl['name'],
48:                 'render_time' => $tpl['render_time'],
49:                 'render_time_str' => $this->formatDuration($tpl['render_time'])
50:             );
51:         }
52: 
53:         return array(
54:             'nb_templates' => count($templates),
55:             'templates' => $templates,
56:             'accumulated_render_time' => $accuRenderTime,
57:             'accumulated_render_time_str' => $this->formatDuration($accuRenderTime)
58:         );
59:     }
60: 
61:     /**
62:      * {@inheritDoc}
63:      */
64:     public function getName()
