143:     /**
144:      * Use this for any adapters that can take connection info in the form below
145:      * to set the adapters connection info.
146:      *
147:      * <code>
148:      * protocol://username:password@host[:port]/dbname
149:      * protocol://urlencoded%20username:urlencoded%20password@host[:port]/dbname?decode=true
150:      * protocol://username:password@unix(/some/file/path)/dbname
151:      * </code>
152:      *
153:      * Sqlite has a special syntax, as it does not need a database name or user authentication:
154:      *
155:      * <code>
156:      * sqlite://file.db
157:      * sqlite://../relative/path/to/file.db
158:      * sqlite://unix(/absolute/path/to/file.db)
159:      * sqlite://windows(c%2A/absolute/path/to/file.db)
160:      * </code>
161:      *
162:      * @param string $connection_url A connection URL
163:      * @return object the parsed URL as an object.
164:      */
165:     public static function parse_connection_url($connection_url)
166:     {
167:         $url = @parse_url($connection_url);
168: 
169:         if (!isset($url['host']))
170:             throw new DatabaseException('Database host must be specified in the connection string. If you want to specify an absolute filename, use e.g. sqlite://unix(/path/to/file)');
171: 
172:         $info = new \stdClass();
173:         $info->protocol = $url['scheme'];
174:         $info->host = $url['host'];
175:         $info->db = isset($url['path']) ? substr($url['path'], 1) : null;
176:         $info->user = isset($url['user']) ? $url['user'] : null;
177:         $info->pass = isset($url['pass']) ? $url['pass'] : null;
178: 
179:         $allow_blank_db = ($info->protocol == 'sqlite');
180: 
181:         if ($info->host == 'unix(')
182:         {
183:             $socket_database = $info->host . '/' . $info->db;
184: 
185:             if ($allow_blank_db)
186:                 $unix_regex = '/^unix\((.+)\)\/?().*$/';
187:             else
188:                 $unix_regex = '/^unix\((.+)\)\/(.+)$/';
189: 
190:             if (preg_match_all($unix_regex, $socket_database, $matches) > 0)
191:             {
192:                 $info->host = $matches[1][0];
193:                 $info->db = $matches[2][0];
194:             }
195:         } elseif (substr($info->host, 0, 8) == 'windows(')
196:         {
197:             $info->host = urldecode(substr($info->host, 8) . '/' . substr($info->db, 0, -1));
198:             $info->db = null;
199:         }
200: 
201:         if ($allow_blank_db && $info->db)
202:             $info->host .= '/' . $info->db;
203: 
204:         if (isset($url['port']))
205:             $info->port = $url['port'];
206: 
207:         if (strpos($connection_url, 'decode=true') !== false)
208:         {
209:             if ($info->user)

211: 
212:             if ($info->pass)
213:                 $info->pass = urldecode($info->pass);
214:         }

216:         if (isset($url['query']))
