23: class SwiftMailCollector extends DataCollector implements Renderable
24: {
25:     protected $messagesLogger;
26: 
27:     public function __construct(Swift_Mailer $mailer)
28:     {
29:         $this->messagesLogger = new Swift_Plugins_MessageLogger();
30:         $mailer->registerPlugin($this->messagesLogger);
31:     }
32: 
33:     /**
34:      * {@inheritDoc}
35:      */
36:     public function collect()
37:     {
38:         $mails = array();
39:         foreach ($this->messagesLogger->getMessages() as $msg) {
40:             $mails[] = array(
41:                 'to' => $this->formatTo($msg->getTo()),
42:                 'subject' => $msg->getSubject(),
43:                 'headers' => $msg->getHeaders()->toString()
44:             );
45:         }
46:         return array(
47:             'count' => count($mails),
48:             'mails' => $mails
49:         );
50:     }
51: 
52:     protected function formatTo($to)
53:     {
54:         $f = array();
55:         foreach ($to as $k => $v) {
56:             $f[] = (empty($v) ? '' : "$v ") . "<$k>";
57:         }
58:         return implode(', ', $f);
59:     }
60: 
61:     /**
62:      * {@inheritDoc}
63:      */
64:     public function getName()
65:     {
66:         return 'swiftmailer_mails';
67:     }
68: 
69:     /**
70:      * {@inheritDoc}
71:      */
72:     public function getWidgets()
73:     {
74:         return array(
75:             'emails' => array(
76:                 'icon' => 'inbox',
77:                 'widget' => 'PhpDebugBar.Widgets.MailsWidget',
78:                 'map' => 'swiftmailer_mails.mails',
79:                 'default' => '[]',
80:                 'title' => 'Mails'
81:             ),
