303:     public static function pluralize( $string )
304:     {
305:         // save some time in the case that singular and plural are the same
306:         if ( in_array( strtolower( $string ), self::$uncountable ) )
307:             return $string;
308: 
309:         // check for irregular singular forms
310:         foreach ( self::$irregular as $pattern => $result )
311:         {
312:             $pattern = '/' . $pattern . '$/i';
313: 
314:             if ( preg_match( $pattern, $string ) )
315:                 return preg_replace( $pattern, $result, $string);
316:         }
317: 
318:         // check for matches using regular expressions
319:         foreach ( self::$plural as $pattern => $result )
320:         {
321:             if ( preg_match( $pattern, $string ) )
