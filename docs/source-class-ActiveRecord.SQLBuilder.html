204:     /**
205:      * Converts a string like "id_and_name_or_z" into a conditions value like array("id=? AND name=? OR z=?", values, ...).
206:      *
207:      * @param Connection $connection
208:      * @param $name Underscored string
209:      * @param $values Array of values for the field names. This is used
210:      *   to determine what kind of bind marker to use: =?, IN(?), IS NULL
211:      * @param $map A hash of "mapped_column_name" => "real_column_name"
212:      * @return A conditions array in the form array(sql_string, value1, value2,...)
213:      */
214:     public static function create_conditions_from_underscored_string(Connection $connection, $name, &$values=array(), &$map=null)
215:     {
216:         if (!$name)
217:             return null;
218: 
219:         $parts = preg_split('/(_and_|_or_)/i',$name,-1,PREG_SPLIT_DELIM_CAPTURE);
220:         $num_values = count($values);
221:         $conditions = array('');
222: 
223:         for ($i=0,$j=0,$n=count($parts); $i<$n; $i+=2,++$j)
224:         {
225:             if ($i >= 2)
226:                 $conditions[0] .= preg_replace(array('/_and_/i','/_or_/i'),array(' AND ',' OR '),$parts[$i-1]);
227: 
228:             if ($j < $num_values)
229:             {
230:                 if (!is_null($values[$j]))
231:                 {
232:                     $bind = is_array($values[$j]) ? ' IN(?)' : '=?';
233:                     $conditions[] = $values[$j];
234:                 }
235:                 else
236:                     $bind = ' IS NULL';
237:             }
238:             else
239:                 $bind = ' IS NULL';
240: 
241:             // map to correct name if $map was supplied
242:             $name = $map && isset($map[$parts[$i]]) ? $map[$parts[$i]] : $parts[$i];
243: 
244:             $conditions[0] .= $connection->quote_name($name) . $bind;
245:         }
246:         return $conditions;
247:     }
