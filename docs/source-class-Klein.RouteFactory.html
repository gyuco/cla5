 41:     /**
 42:      * Check if the path is null or equal to our match-all, null-like value
 43:      *
 44:      * @param mixed $path
 45:      * @access protected
 46:      * @return boolean
 47:      */
 48:     protected function pathIsNull($path)
 49:     {
 50:         return (static::NULL_PATH_VALUE === $path || null === $path);
 51:     }
 52: 
 53:     /**
 54:      * Quick check to see whether or not to count the route
 55:      * as a match when counting total matches
 56:      *
 57:      * @param string $path
 58:      * @access protected
 59:      * @return boolean
 60:      */
 61:     protected function shouldPathStringCauseRouteMatch($path)
 62:     {
 63:         // Only consider a request to be matched when not using 'matchall'
 64:         return !$this->pathIsNull($path);
 65:     }
 66: 
 67:     /**
 68:      * Pre-process a path string
 69:      *
 70:      * This method wraps the path string in a regular expression syntax baesd
 71:      * on whether the string is a catch-all or custom regular expression.
 72:      * It also adds the namespace in a specific part, based on the style of expression
 73:      *
 74:      * @param string $path
 75:      * @access protected
 76:      * @return string
 77:      */
 78:     protected function preprocessPathString($path)
 79:     {
 80:         // If the path is null, make sure to give it our match-all value
 81:         $path = (null === $path) ? static::NULL_PATH_VALUE : (string) $path;
 82: 
 83:         // If a custom regular expression (or negated custom regex)
 84:         if ($this->namespace && $path[0] === '@' || ($path[0] === '!' && $path[1] === '@')) {
 85:             // Is it negated?
 86:             if ($path[0] === '!') {
 87:                 $negate = true;
 88:                 $path = substr($path, 2);
 89:             } else {
 90:                 $negate = false;
 91:                 $path = substr($path, 1);
 92:             }
 93: 
 94:             // Regex anchored to front of string
 95:             if ($path[0] === '^') {
 96:                 $path = substr($path, 1);
 97:             } else {
 98:                 $path = '.*' . $path;
 99:             }
100: 
101:             if ($negate) {
102:                 $path = '@^' . $this->namespace . '(?!' . $path . ')';
103:             } else {
104:                 $path = '@^' . $this->namespace . $path;
105:             }
106: 
107:         } elseif ($this->namespace && $this->pathIsNull($path)) {
108:             // Empty route with namespace is a match-all
109:             $path = '@^' . $this->namespace . '(/|$)';
110:         } else {
111:             // Just prepend our namespace
112:             $path = $this->namespace . $path;
113:         }
114: 
115:         return $path;
116:     }
117:

119:      * Build a Route instance
120:      *
121:      * @param callable $callback    Callable callback method to execute on route match
122:      * @param string $path          Route URI path to match
123:      * @param string|array $method  HTTP Method to match
124:      * @param boolean $count_match  Whether or not to count the route as a match when counting total matches
125:      * @param string $name          The name of the route
126:      * @static
127:      * @access public
128:      * @return Route
129:      */
130:     public function build($callback, $path = null, $method = null, $count_match = true, $name = null)
131:     {
132:         return new Route(
133:             $callback,
134:             $this->preprocessPathString($path),
135:             $method,
136:             $this->shouldPathStringCauseRouteMatch($path) // Ignore the $count_match boolean that they passed
137:         );
138:     }
