 33:     public function onBootstrap(EventInterface $event)
 34:     {
 35:         $prettyPageHandler = new PrettyPageHandler();
 36: 
 37:         // Set editor
 38:         $config = $event->getApplication()->getServiceManager()->get('Config');
 39:         if (isset($config['view_manager']['editor'])) {
 40:             $prettyPageHandler->setEditor($config['view_manager']['editor']);
 41:         }
 42: 
 43: 
 44:         $this->run = new Run();
 45:         $this->run->register();
 46:         $this->run->pushHandler($prettyPageHandler);
 47: 
 48:         $this->attachListeners($event);
 49:     }
 50: 
 51:     public function getAutoloaderConfig()
 52:     {
 53:         return array(
 54:             'Zend\Loader\StandardAutoloader' => array(
 55:                 'namespaces' => array(
 56:                     __NAMESPACE__ => __DIR__ . '/src/' . __NAMESPACE__,
 57:                 ),
 58:             ),
 59:         );
 60:     }
 61: 
 62:     private function attachListeners(EventInterface $event)
 63:     {
 64:         $request = $event->getRequest();
 65:         $application = $event->getApplication();
 66:         $services = $application->getServiceManager();
 67:         $events = $application->getEventManager();
 68:         $config = $services->get('Config');
 69: 
 70:         //Display exceptions based on configuration and console mode
 71:         if ($request instanceof ConsoleRequest || empty($config['view_manager']['display_exceptions']))
 72:             return;
 73: 
 74:         $jsonHandler = new JsonResponseHandler();
 75: 
 76:         if (!empty($config['view_manager']['json_exceptions']['show_trace'])) {
 77:             //Add trace to the JSON output
 78:             $jsonHandler->addTraceToOutput(true);
 79:         }
 80: 
 81:         if (!empty($config['view_manager']['json_exceptions']['ajax_only'])) {
 82:             //Only return JSON response for AJAX requests
 83:             $jsonHandler->onlyForAjaxRequests(true);
 84:         }
 85: 
 86:         if (!empty($config['view_manager']['json_exceptions']['display'])) {
 87:             //Turn on JSON handler
 88:             $this->run->pushHandler($jsonHandler);
 89:         }
 90: 
 91:         //Attach the Whoops ExceptionStrategy
 92:         $exceptionStrategy = new ExceptionStrategy($this->run);
 93:         $exceptionStrategy->attach($events);
 94: 
 95:         //Attach the Whoops RouteNotFoundStrategy
 96:         $routeNotFoundStrategy = new RouteNotFoundStrategy($this->run);
 97:         $routeNotFoundStrategy->attach($events);
 98: 
 99:         //Detach default ExceptionStrategy
100:         $services->get('Zend\Mvc\View\Http\ExceptionStrategy')->detach($events);
101: 
102:         //Detach default RouteNotFoundStrategy
103:         $services->get('Zend\Mvc\View\Http\RouteNotFoundStrategy')->detach($events);
104:     }
